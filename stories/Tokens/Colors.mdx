import { Meta, ColorPalette, ColorItem } from '@storybook/blocks';

import { Color } from '~/assets/scripts/tokens.ts'

import { capitalize } from '~/assets/scripts/utils.ts';

<Meta title="Tokens/Colors" />

# Color System

## Theme

{Object.entries(Color.theme.default).map((name, i) => {
  return (
    <>
      <h3>{capitalize(name[0])}</h3>
      <ColorPalette>
      {Object.entries(name[1]).map((category, i) => {
        return (
          <ColorItem
            key={category[0]}
            title={capitalize(category[0])}
            colors={{...category[1]}}
          />
        )
      })}
      </ColorPalette>
    </>
  )})
}

## Palette

<ColorPalette>
{Object.entries(Color.palette.default).map((color, i) => {
  return (
    <ColorItem
      key={color[0]}
      title={capitalize(color[0])}
      colors={{ [color[0]]: color[1] }}
    />
  )})
}
</ColorPalette>
